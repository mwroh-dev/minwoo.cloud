```python {2, 4-5} showLineNumbers
from typing import List

def build_lps(pattern: str) -> List[int]:
	lps = [0] * len(pattern)
	length = 0
	i = 1

	while i < len(pattern):
		if pattern[i] == pattern[length]:
			length += 1
			lps[i] = length
			i += 1
		elif length > 0:
			length = lps[length - 1]
		else:
			lps[i] = 0
			i += 1

	return lps
```
